   # Inside psql:

   # List all cost events
   SELECT id, api_provider_id, model_id, user_id, input_tokens,
   output_tokens,
          cost_usd, event_timestamp FROM cost_events ORDER BY id DESC LIMIT
    10;

   # Count events by provider
   SELECT p.name, COUNT\(*\) FROM cost_events ce
     JOIN models m ON ce.model_id = m.id
     JOIN api_providers p ON m.api_provider_id = p.id
     GROUP BY p.name;

   # Total cost by user
   SELECT user_id, SUM\(cost_usd\) as total_cost FROM cost_events
     GROUP BY user_id ORDER BY total_cost DESC;

   # Check for duplicates \(should be 0 for duplicates we detected\)
   SELECT COUNT\(*\) FROM cost_events
     WHERE request_id IN \(SELECT request_id FROM cost_events GROUP BY
   request_id HAVING COUNT\(*\) > 1\);
